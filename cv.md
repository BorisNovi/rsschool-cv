function narcissistic(value) {
    // Code me to return true or false
    expo = [];
    numOfExpo = (value + "").length;
    (value + "").split('').forEach((item) => expo.push(item ** numOfExpo));
    expo = expo.reduce((sum, current) => sum + current);

    return value == expo
};
